###Question 8: Follow the instructions from the UNDP’s “Technical Note 1” to calculate the component indexes of the HDI and then the HDI itself.

#The life expectancy index (call it HDI$lifexpI).
HDI$lifexpI <- (HDI$lifexp - 20) / (85 - 20)
summary(HDI$lifexpI)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##  0.4956  0.7194  0.8269  0.8024  0.8827  0.9861
#The education index (call it HDI$educI);
school.expI <- (HDI$school.exp - 0) / (18 - 0)
school.meanI <- (HDI$school.mean - 0) / (15 - 0)
HDI$educI <- (school.meanI + school.expI) / 2
summary(HDI$educI)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##  0.2142  0.5144  0.6797  0.6532  0.7902  1.0652
#The income index (call it HDI$incomeI);
HDI[HDI$gni.pc > 75000 & !is.na(HDI$gni.pc), "gni.pc"] <- 75000
HDI$incomeI <- (log(HDI$gni.pc) - log(100)) / (log(75000) - log(100))
summary(HDI$incomeI)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##  0.2858  0.5512  0.7114  0.6925  0.8364  1.0000
#The overall HDI (call it HDI$myHDI)
HDI$myHDI <- (HDI$lifexpI * HDI$educI * HDI$incomeI)^(1/3)
summary(HDI$myHDI)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##  0.3539  0.5899  0.7354  0.7098  0.8210  0.9826
